Reading the moisture level in a plant and displaying it on a web page

def update_leds():
    for i in range(20):
        moisture = read_moisture()
        if moisture < 700:
            color = (255, int(moisture / 10), 0)
        else:
            color = (int(255 - moisture / 10), 255, 0)
